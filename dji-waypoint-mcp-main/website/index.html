快速开始

1

环境准备

# 确保Python 3.8+环境
python --version

# 克隆项目
git clone <repository-url>
cd dji-waypoint-mcp
2

安装依赖

# 安装依赖
pip install -r requirements.txt

# 安装开发依赖（可选）
pip install -r requirements-dev.txt
3

配置MCP

# 在.kiro/settings/mcp.json中添加配置
{
  "mcpServers": {
    "dji-waypoint-mcp": {
      "command": "python",
      "args": ["run_server.py"],
      "env": {
        "PYTHONPATH": ".",
        "LOG_LEVEL": "INFO"
      },
      "disabled": false
    }
  }
}
4

启动服务

# 启动MCP服务
python run_server.py

# 或者直接运行模块
python -m src.dji_waypoint_mcp.server
